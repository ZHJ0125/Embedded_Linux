	DIR* dp;

		printf("Usage ex3_9 <pathname>\n");
		return -1;
	}
	// Open directory

		printf("Opendir %s failed!\n",argv[1]);
		return -2;
	}
	// Read folder data
	if((pdirent = readdir(dp)) == 0){
		printf("Fail to call readdir()\n");
		return -3;
	}
	// Output folder data
	printf("In directory %s has following files:\n%s\n",argv[1],pdirent->d_name);
	// Close directory
	closedir(dp);
	return 0;
}

struct student{
    char name[NAMESIZE];
    long number;
    short department;
    short scores[10];

int main(int argc, char* argv[]){
	int num;
	int subject;
	FILE* fpstudents;
	// Init data
	strcpy(stu[0].name, "zhang0");
	strcpy(stu[1].name, "zhang1");

			stu[num].scores[subject] = num * subject + 60;
		}

	// Check parameters
    if(argc <= 1){

        printf("Open sourcefile %s failed!\n",argv[1]);
